function buyWithTon(itemId, price) { // Вибрация при нажатии if (tg.HapticFeedback) tg.HapticFeedback.impactOccurred('heavy'); // Стандартное окно подтверждения в Telegram tg.showConfirm(`Вы подтверждаете покупку "${itemId}" за ${price} TON?`, (ok) => { if (ok) { // Отправка данных боту. Бот должен поймать это через обработчик данных tg.sendData(JSON.stringify({ action: "buy_ton", item: itemId, price: price })); closeAllModals(); tg.showAlert("Запрос на оплату отправлен боту! Проверьте чат."); } }); }
