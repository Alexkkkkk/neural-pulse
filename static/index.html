// --- ИСПРАВЛЕННАЯ ЛОГИКА ЗАГРУЗКИ --- function startLoadingAnimation() { let progress = 0; const percentEl = document.getElementById('loader-percent'); const fillEl = document.getElementById('loader-fill'); const statusEl = document.querySelector('.loader-text'); const statuses = ["SCANNING IDENTITY...", "DECRYPTING LINK...", "ESTABLISHING SYNC...", "ACCESS GRANTED"]; // ПРЕДОХРАНИТЕЛЬ: Если через 7 секунд данные не пришли, открываем игру принудительно const safetyTimer = setTimeout(() => { console.log("Safety trigger: forcing load"); state.loaded = true; }, 7000); const interval = setInterval(() => { if (progress < 25) statusEl.innerText = statuses[0]; else if (progress < 50) statusEl.innerText = statuses[1]; else if (progress < 80) statusEl.innerText = statuses[2]; else statusEl.innerText = statuses[3]; // Условие выхода: прогресс 100% И подтверждение загрузки данных if (progress >= 100 && state.loaded) { clearInterval(interval); clearTimeout(safetyTimer); // Убираем таймер, если всё загрузилось само tg.HapticFeedback.notificationOccurred('success'); setTimeout(() => { document.getElementById('loader-screen').style.opacity = '0'; document.querySelector('.content-layer').classList.add('active'); setTimeout(() => document.getElementById('loader-screen').style.display = 'none', 850); }, 300); return; } // Замедляем полоску на 99%, если данные еще не получены if (progress < 99) { progress += Math.floor(Math.random() * 5) + 2; } else if (state.loaded) { progress = 100; } if (progress > 100) progress = 100; percentEl.innerText = `${progress}%`; fillEl.style.width = `${progress}%`; }, 80); } // --- ИСПРАВЛЕННЫЙ ЗАПРОС К СЕРВЕРУ --- async function loadData() { if(!state.uid) { state.loaded = true; return; } try { const r = await fetch(`/api/balance/${state.uid}`); if (!r.ok) throw new Error('Server error'); // Проверка на 404/500 ошибки const res = await r.json(); if(res.status === 'ok') { state.balance = parseFloat(res.data.balance || 0); state.click_lvl = parseInt(res.data.click_lvl || 1); state.pnl = parseFloat(res.data.pnl || 0); state.energy = parseFloat(res.data.energy || 1000); state.max_energy = parseInt(res.data.max_energy || 1000); updateUI(); } } catch(e) { console.error("Load failed, using default state:", e); } finally { // ЭТО КЛЮЧЕВОЙ МОМЕНТ: // loaded станет true даже если сервер выключен или выдал ошибку. state.loaded = true; } }
